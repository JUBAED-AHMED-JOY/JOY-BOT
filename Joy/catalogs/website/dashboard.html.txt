          <button class="btn btn-sm btn-success" onclick="addUID('ADMINBOT')"><i class="fas fa-plus"></i></button>
        </div>
      </div>
      <div>
        <div class="cfg-label" style="margin-bottom:8px;color:#8b5cf6;">⚙️ OPERATOR UIDs</div>
        <div class="uid-box" id="uid_OPERATOR"></div>
        <div style="display:flex;gap:6px;margin-top:8px;">
          <input class="cfg-input" type="text" id="uid_OPERATOR_input" placeholder="UID e.g. 100000..." style="flex:1;">
          <button class="btn btn-sm btn-success" onclick="addUID('OPERATOR')"><i class="fas fa-plus"></i></button>
        </div>
      </div>
    </div>
  </div>

  <!-- Commands -->
  <div class="card grid-bottom">
    <div style="display:flex;align-items:center;justify-content:space-between;margin-bottom:14px;flex-wrap:wrap;gap:10px;">
      <div class="card-title" style="margin:0;"><i class="fas fa-puzzle-piece"></i> Installed Commands <span class="badge badge-blue" id="cmdCount">0</span></div>
    </div>
    <input class="cmd-search" type="text" id="cmdSearch" placeholder="🔍  Search commands..." oninput="filterCommands()">
    <div class="cmd-grid" id="cmdGrid">
      <div style="color:var(--muted);font-size:13px;grid-column:1/-1;text-align:center;padding:30px 0;">
        <i class="fas fa-spinner fa-spin" style="margin-right:8px;"></i> Loading commands...
      </div>
    </div>
  </div>

</div>

<!-- Toast -->
<div class="toast" id="toast">
  <i id="toastIcon" class="fas fa-check-circle"></i>
  <span id="toastMsg">Success</span>
</div>

<script>
  // ── State ──────────────────────────────────────────────────
  let lastLogTime = 0;
  let allCommands = [];
  let fileContent = null;

  // ── Toast ──────────────────────────────────────────────────
  function showToast(msg, type = 'success') {
    const t = document.getElementById('toast');
    const icon = document.getElementById('toastIcon');
    document.getElementById('toastMsg').textContent = msg;
    t.className = 'toast show ' + type;
    icon.className = type === 'success' ? 'fas fa-check-circle' : 'fas fa-times-circle';
    icon.style.color = type === 'success' ? '#10b981' : '#ef4444';
    clearTimeout(t._timeout);
    t._timeout = setTimeout(() => t.className = 'toast', 3000);
  }

  // ── Tab ────────────────────────────────────────────────────
  function switchTab(id, btn) {
    document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    document.getElementById('tab-' + id).classList.add('active');
    btn.classList.add('active');
  }

  // ── Status polling ─────────────────────────────────────────
  async function pollStatus() {
    try {
      const r = await fetch('/api/status');
      const d = await r.json();

      const dot = document.getElementById('headerDot');
      const hs = document.getElementById('headerStatus');
      const ss = document.getElementById('statStatus');
      const ssu = document.getElementById('statServerUptime');
      const sbu = document.getElementById('statBotUptime');
      const sc = document.getElementById('statCmds');
      const sr = document.getElementById('statRestarts');
      const sa = document.getElementById('statAppstate');

      const online = d.status === 'running';
      dot.className = 'status-dot ' + (online ? 'online' : 'offline');
      hs.textContent = online ? 'Online' : d.status === 'starting' ? 'Starting...' : 'Offline';
      ss.textContent = d.status.charAt(0).toUpperCase() + d.status.slice(1);
      ss.style.color = online ? '#10b981' : '#ef4444';
      ssu.textContent = formatUptime(d.serverUptime);
      sbu.textContent = formatUptime(d.botUptime);
      sc.textContent = d.commands;
      sr.textContent = d.restarts;
      sa.innerHTML = d.joystateExists
        ? '<span style="color:#10b981;font-size:18px;">✔</span>'
        : '<span style="color:#ef4444;font-size:18px;">✘</span>';
    } catch (_) {}
  }

  function formatUptime(s) {
    if (s < 60) return s + 's';
    if (s < 3600) return Math.floor(s / 60) + 'm ' + (s % 60) + 's';
    const h = Math.floor(s / 3600);
    const m = Math.floor((s % 3600) / 60);
    return h + 'h ' + m + 'm';
  }

  // ── Log polling ────────────────────────────────────────────
  async function pollLogs() {
    try {
      const r = await fetch('/api/logs?since=' + lastLogTime);
      const lines = await r.json();
      if (!lines.length) return;
      const box = document.getElementById('consoleBox');
      const autoScroll = document.getElementById('autoScrollChk').checked;
      lines.forEach(l => {
        lastLogTime = Math.max(lastLogTime, l.t);
        const div = document.createElement('div');
        div.className = 'log-line';
        const t = new Date(l.t).toLocaleTimeString();
        let cls = 'log-msg';
        const m = l.m.toLowerCase();
        if (m.includes('error') || m.includes('[ error ]')) cls += ' err';
        else if (m.includes('warn')) cls += ' warn';
        else if (m.includes('success') || m.includes('deployed') || m.includes('✅') || m.includes('running')) cls += ' ok';
        else if (m.includes('[ system ]') || m.includes('[ server ]') || m.includes('[ dashboard ]')) cls += ' info';
        div.innerHTML = `<span class="log-time">${t}</span><span class="${cls}">${escHtml(l.m)}</span>`;
        box.appendChild(div);
      });
      if (autoScroll) box.scrollTop = box.scrollHeight;
    } catch (_) {}
  }

  function escHtml(s) {
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  }

  function clearConsole() {
    document.getElementById('consoleBox').innerHTML = '';
  }

  // ── Restart ────────────────────────────────────────────────
  async function restartBot() {
    try {
      await fetch('/api/restart', { method: 'POST' });
      showToast('Bot restart initiated!');
    } catch (_) {
      showToast('Failed to restart', 'error');
    }
  }

  // ── Joystate ───────────────────────────────────────────────
  async function saveJoystate() {
    const raw = document.getElementById('joystateInput').value.trim();
    if (!raw) return showToast('Please paste your appstate JSON first', 'error');
    let parsed;
    try { parsed = JSON.parse(raw); } catch (e) { return showToast('Invalid JSON: ' + e.message, 'error'); }
    try {
      const r = await fetch('/api/joystate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(parsed)
      });
      const d = await r.json();
      if (d.ok) { showToast('Appstate saved! Bot restarting...'); document.getElementById('joystateInput').value = ''; }
      else showToast(d.error || 'Failed to save', 'error');
    } catch (_) { showToast('Network error', 'error'); }
  }

  function handleFileUpload(e) {
    const file = e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = ev => {
      try {
        fileContent = JSON.parse(ev.target.result);
        document.getElementById('filePreview').textContent = '✔ File loaded: ' + file.name + ' (' + (Array.isArray(fileContent) ? fileContent.length + ' cookies' : 'object') + ')';
        document.getElementById('filePreview').style.color = '#10b981';
      } catch (err) {
        fileContent = null;
        document.getElementById('filePreview').textContent = '✘ Invalid JSON: ' + err.message;
        document.getElementById('filePreview').style.color = '#ef4444';
      }
    };
    reader.readAsText(file);
  }

  async function saveJoystateFile() {
    if (!fileContent) return showToast('Please select a valid Joystate.json file first', 'error');
    try {
      const r = await fetch('/api/joystate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(fileContent)
      });
      const d = await r.json();
      if (d.ok) { showToast('Appstate saved! Bot restarting...'); fileContent = null; document.getElementById('filePreview').textContent = ''; document.getElementById('fileInput').value = ''; }
      else showToast(d.error || 'Failed to save', 'error');
    } catch (_) { showToast('Network error', 'error'); }
  }

  // Drag & drop
  const dz = document.getElementById('dropZone');
  dz.addEventListener('dragover', e => { e.preventDefault(); dz.style.borderColor = '#3b82f6'; });
  dz.addEventListener('dragleave', () => { dz.style.borderColor = ''; });
  dz.addEventListener('drop', e => {
    e.preventDefault(); dz.style.borderColor = '';
    const file = e.dataTransfer.files[0];
    if (file) { const dt = new DataTransfer(); dt.items.add(file); document.getElementById('fileInput').files = dt.files; handleFileUpload({ target: { files: [file] } }); }
  });

  // ── Commands ───────────────────────────────────────────────
  async function loadCommands() {
    try {
      const r = await fetch('/api/commands');
      allCommands = await r.json();
      renderCommands(allCommands);
    } catch (_) {}
  }

  function renderCommands(cmds) {
    const grid = document.getElementById('cmdGrid');
    document.getElementById('cmdCount').textContent = cmds.length;
    if (!cmds.length) {
      grid.innerHTML = '<div style="color:var(--muted);font-size:13px;grid-column:1/-1;text-align:center;padding:30px 0;">No commands found.</div>';
      return;
    }
    grid.innerHTML = cmds.map(c => `
      <div class="cmd-item">
        <div class="cmd-name">.${c.name}</div>
        <div class="cmd-cat">${c.category}</div>
        <div class="cmd-desc">${escHtml(c.description)}</div>
      </div>
    `).join('');
  }

  function filterCommands() {
    const q = document.getElementById('cmdSearch').value.toLowerCase();
    renderCommands(allCommands.filter(c =>
      c.name.toLowerCase().includes(q) ||
      c.category.toLowerCase().includes(q) ||
      c.description.toLowerCase().includes(q)
    ));
  }

  // ── Joy.json Config ────────────────────────────────────────
  const uidData = { OWNER: [], ADMINBOT: [], OPERATOR: [] };

  function renderUIDs(key) {
    const box = document.getElementById('uid_' + key);
    if (!box) return;
    box.innerHTML = uidData[key].length === 0
      ? '<span style="color:var(--muted);font-size:12px;">No UIDs added</span>'
      : uidData[key].map((uid, i) => `
        <span class="uid-tag">
          ${escHtml(uid)}
          <button onclick="removeUID('${key}',${i})" title="Remove">✕</button>
        </span>`).join('');
  }

  function addUID(key) {
    const input = document.getElementById('uid_' + key + '_input');
    const val = input.value.trim();
    if (!val) return;
    if (!/^\d+$/.test(val)) return showToast('UID must be numbers only', 'error');
    if (uidData[key].includes(val)) return showToast('UID already added', 'error');
    uidData[key].push(val);
    input.value = '';
    renderUIDs(key);
  }

  function removeUID(key, idx) {
    uidData[key].splice(idx, 1);
    renderUIDs(key);
  }

  // Enter key support for UID inputs
  ['OWNER','ADMINBOT','OPERATOR'].forEach(k => {
    document.addEventListener('DOMContentLoaded', () => {
      const inp = document.getElementById('uid_' + k + '_input');
      if (inp) inp.addEventListener('keydown', e => { if (e.key === 'Enter') addUID(k); });
    });
  });

  async function loadConfig() {
    try {
      const r = await fetch('/api/joyconfig');
      const d = await r.json();

      // Basic fields
      ['BOTNAME','PREFIX','EMOJI'].forEach(k => {
        const el = document.getElementById('cfg_' + k);
        if (el) el.value = d[k] || '';
      });

      // Language
      const lang = document.getElementById('cfg_language');
      if (lang) lang.value = d.language || 'en';

      // Toggles
      ['autoRestart','saveState','approval','notification','banMsg','adminOnlyMsg','accpetPending','developermode'].forEach(k => {
        const el = document.getElementById('cfg_' + k);
        if (el) el.checked = !!d[k];
      });

      // UID arrays
      ['OWNER','ADMINBOT','OPERATOR'].forEach(k => {
        uidData[k] = Array.isArray(d[k]) ? [...d[k]] : [];
        renderUIDs(k);
      });
    } catch (e) {
      showToast('Failed to load config', 'error');
    }
  }

  async function saveConfig() {
    const payload = {
      BOTNAME: document.getElementById('cfg_BOTNAME').value.trim(),
      PREFIX:  document.getElementById('cfg_PREFIX').value.trim() || '.',
      EMOJI:   document.getElementById('cfg_EMOJI').value.trim(),
      language: document.getElementById('cfg_language').value,
      autoRestart:    document.getElementById('cfg_autoRestart').checked,
      saveState:      document.getElementById('cfg_saveState').checked,
      approval:       document.getElementById('cfg_approval').checked,
      notification:   document.getElementById('cfg_notification').checked,
      banMsg:         document.getElementById('cfg_banMsg').checked,
      adminOnlyMsg:   document.getElementById('cfg_adminOnlyMsg').checked,
      accpetPending:  document.getElementById('cfg_accpetPending').checked,
      developermode:  document.getElementById('cfg_developermode').checked,
      OWNER:    uidData.OWNER,
      ADMINBOT: uidData.ADMINBOT,
      OPERATOR: uidData.OPERATOR,
    };
    if (!payload.BOTNAME) return showToast('Bot Name cannot be empty', 'error');
    try {
      const r = await fetch('/api/joyconfig', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      const d = await r.json();
      if (d.ok) showToast('Config saved! Bot restarting...');
      else showToast(d.error || 'Failed to save', 'error');
    } catch (_) { showToast('Network error', 'error'); }
  }

  // ── Init ───────────────────────────────────────────────────
  pollStatus();
  pollLogs();
  loadCommands();
  loadConfig();
  setInterval(pollStatus, 3000);
  setInterval(pollLogs, 1500);

  // UID input enter key (after DOM ready)
  ['OWNER','ADMINBOT','OPERATOR'].forEach(k => {
    const inp = document.getElementById('uid_' + k + '_input');
    if (inp) inp.addEventListener('keydown', e => { if (e.key === 'Enter') addUID(k); });
  });
</script>
</body>
  </html>
